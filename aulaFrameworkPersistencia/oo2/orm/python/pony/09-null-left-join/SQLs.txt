-- técnicos vinculadas a serviços
SELECT p.nome
FROM Pessoa p 
JOIN Tecnico t on p.id = t.pessoa 
JOIN Servico s on s.tecnico = t.id 

-- técnicos e serviços
SELECT p.nome, s.id, s.descricao 
FROM Pessoa p
JOIN Tecnico t on p.id = t.pessoa
LEFT JOIN Servico s on s.tecnico = t.id 

-- técnicos que não realizaram nenhum serviço
SELECT p.nome
FROM Pessoa p
JOIN Tecnico t on p.id = t.pessoa
LEFT JOIN Servico s on s.tecnico = t.id 
WHERE s.id IS NULL


-- técnicos e serviços realizados por eles
SELECT p.nome, s.descricao
FROM Pessoa p, Tecnico t, Servico s
WHERE p.id = t.pessoa and s.tecnico = t.id 

-- técnicos que não realizaram nenhum serviço (outra forma)
SELECT p.nome
FROM Pessoa p, Tecnico t
WHERE p.id = t.pessoa 
AND p.nome NOT IN (
SELECT nome
FROM Pessoa p, Tecnico t, Servico s
WHERE p.id = t.pessoa and s.tecnico = t.id)

-- quantidade de serviços realizados pelos técnicos
SELECT p.nome, count(s.id) q
FROM Pessoa p
JOIN Tecnico t on p.id = t.pessoa
LEFT JOIN Servico s on s.tecnico = t.id
GROUP BY p.nome

-- valor orçado para cada cliente, total

FROM Pessoa p
JOIN Cliente c on p.id=c.pessoa 
JOIN Servico s on s.cliente =c.id 
GROUP BY p.nome
