@startuml
!theme plain
hide circle
hide methods
hide stereotypes
skinparam linetype ortho

entity "Pessoa" as Pessoa {
  *id : INTEGER <<PK>>
  --
  nome : TEXT
  email : TEXT <<UNIQUE>>
  telefone : TEXT
}

entity "Categoria" as Categoria {
  *id : INTEGER <<PK>>
  --
  nome : TEXT <<UNIQUE>>
  descricao : TEXT
}

entity "Receita" as Receita {
  *id : INTEGER <<PK>>
  --
  titulo : TEXT
  descricao : TEXT
  modo_preparo : TEXT
  tempo_preparo : INTEGER
  rendimento : TEXT
  autor : INTEGER <<FK>>
  categoria : INTEGER <<FK>>
}

entity "Ingrediente" as Ingrediente {
  *id : INTEGER <<PK>>
  --
  nome : TEXT <<UNIQUE>>
  unidade_padrao : TEXT
}

entity "ItemIngrediente" as ItemIngrediente {
  *id : INTEGER <<PK>>
  --
  receita : INTEGER <<FK>>
  ingrediente : INTEGER <<FK>>
  quantidade : REAL
  unidade : TEXT
}

entity "Comentario" as Comentario {
  *id : INTEGER <<PK>>
  --
  pessoa : INTEGER <<FK>>
  receita : INTEGER <<FK>>
  texto : TEXT
  data : TEXT
  nota : INTEGER
}

' =======================
' RELACIONAMENTOS
' =======================

' Pessoa -> Receita (autor)
Pessoa ||--o{ Receita : "1..N (cria / é autor de)"

' Pessoa -> Comentario
Pessoa ||--o{ Comentario : "1..N (escreve)"

' Categoria -> Receita
Categoria ||--o{ Receita : "1..N (classifica)"

' Receita -> ItemIngrediente
Receita ||--o{ ItemIngrediente : "1..N (contém)"

' Ingrediente -> ItemIngrediente
Ingrediente ||--o{ ItemIngrediente : "1..N (é usado em)"

' Receita -> Comentario
Receita ||--o{ Comentario : "1..N (recebe)"

@enduml
