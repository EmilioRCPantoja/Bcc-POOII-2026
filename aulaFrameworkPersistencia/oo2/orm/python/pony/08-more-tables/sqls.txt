-- em quanta receitas cada ingrediente foi usado?

SELECT count(ii.ingrediente) n, i.nome ingrediente
FROM Ingrediente i, ItemIngrediente ii, Receita r 
WHERE  i.id = ii.ingrediente AND 
r.id  = ii.receita 
GROUP BY ingrediente
ORDER BY n desc, ingrediente

-- qual é a nota média por receita?

SELECT avg(c.nota) media, r.titulo receita
FROM Receita r, Comentario c 
WHERE r.id = c.receita 
group by receita

-- receitas por ordem de tempo de preparo
SELECT titulo, tempo_preparo
FROM Receita 
ORDER BY tempo_preparo 

-- receitas e quantidade de ingredientes em cada receita
SELECT count(ii.id) q, r.titulo receita
FROM Receita r, ItemIngrediente ii 
WHERE r.id  = ii.receita 
GROUP BY receita

-- o quanto de cada ingrediente é preciso para fazer cada receita
SELECT sum(ii.quantidade) q, i.nome ingrediente
FROM Ingrediente i, Receita r, ItemIngrediente ii
WHERE
i.id = ii.ingrediente AND
r.id = ii.receita 
GROUP BY ingrediente
ORDER BY q DESC


SELECT sum(ii.quantidade) q, i.nome ingrediente, r.titulo 
FROM Ingrediente i, Receita r, ItemIngrediente ii
WHERE
i.id = ii.ingrediente AND
r.id = ii.receita
GROUP BY ingrediente, r.titulo 
ORDER BY r.titulo DESC

-- comentários e receitas
SELECT c.texto, r.titulo
FROM Comentario c, Receita r 
WHERE
c.receita = r.id 



-- receitas mais demoradas
SELECT r.titulo, r.tempo_preparo 
FROM Receita r 
ORDER BY r.tempo_preparo DESC	


CREATE VIEW ReceitasTempos(receita, tempo) AS
SELECT r.titulo, r.tempo_preparo 
FROM Receita r 
ORDER BY r.tempo_preparo DESC	



CREATE VIEW NotasReceitas (nota, receita) AS
SELECT avg(c.nota), r.titulo receita
FROM Receita r, Comentario c 
WHERE r.id = c.receita 
group by receita

-- receitas  nas quais um ingrediente está em 
-- maior quantidade do que na outra
SELECT i.nome, ii.ingrediente, ii.receita, ii.quantidade, 
       ii2.receita, ii2.quantidade   
FROM   ItemIngrediente ii, ItemIngrediente ii2, Ingrediente i  
WHERE  ii.quantidade > ii2.quantidade 
and    ii.ingrediente = ii2.ingrediente 
and    i.id = ii.ingrediente 
and    i.id = ii2.ingrediente
