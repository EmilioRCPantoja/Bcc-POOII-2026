
-- Categoria definition

CREATE TABLE "Categoria" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "nome" TEXT UNIQUE NOT NULL,
  "descricao" TEXT NOT NULL
);

-- Comentario definition

CREATE TABLE "Comentario" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "pessoa" INTEGER NOT NULL REFERENCES "Pessoa" ("id") ON DELETE CASCADE,
  "receita" INTEGER NOT NULL REFERENCES "Receita" ("id") ON DELETE CASCADE,
  "texto" TEXT NOT NULL,
  "data" TEXT NOT NULL,
  "nota" INTEGER
);

CREATE INDEX "idx_comentario__pessoa" ON "Comentario" ("pessoa");
CREATE INDEX "idx_comentario__receita" ON "Comentario" ("receita");

-- Ingrediente definition

CREATE TABLE "Ingrediente" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "nome" TEXT UNIQUE NOT NULL,
  "unidade_padrao" TEXT NOT NULL
);

-- ItemIngrediente definition

CREATE TABLE "ItemIngrediente" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "receita" INTEGER NOT NULL REFERENCES "Receita" ("id") ON DELETE CASCADE,
  "ingrediente" INTEGER NOT NULL REFERENCES "Ingrediente" ("id") ON DELETE CASCADE,
  "quantidade" REAL NOT NULL,
  "unidade" TEXT NOT NULL
);

CREATE INDEX "idx_itemingrediente__ingrediente" ON "ItemIngrediente" ("ingrediente");
CREATE INDEX "idx_itemingrediente__receita" ON "ItemIngrediente" ("receita");

-- Pessoa definition

CREATE TABLE "Pessoa" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "nome" TEXT NOT NULL,
  "email" TEXT UNIQUE NOT NULL,
  "telefone" TEXT NOT NULL
);

-- Receita definition

CREATE TABLE "Receita" (
  "id" INTEGER PRIMARY KEY AUTOINCREMENT,
  "titulo" TEXT NOT NULL,
  "descricao" TEXT NOT NULL,
  "modo_preparo" TEXT NOT NULL,
  "tempo_preparo" INTEGER,
  "rendimento" TEXT NOT NULL,
  "autor" INTEGER NOT NULL REFERENCES "Pessoa" ("id") ON DELETE CASCADE,
  "categoria" INTEGER REFERENCES "Categoria" ("id") ON DELETE SET NULL
);

CREATE INDEX "idx_receita__autor" ON "Receita" ("autor");
CREATE INDEX "idx_receita__categoria" ON "Receita" ("categoria");
