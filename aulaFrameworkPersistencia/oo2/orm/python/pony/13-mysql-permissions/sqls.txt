mariasilva / msilva
- pode acessar tabela de pessoas e alterar dados (update)

CREATE USER 'mariasilva'@'%' IDENTIFIED BY 'msilva';

GRANT SELECT ON `hylson`.`hylson_emprestimo` TO `mariasilva`@`%`;
GRANT SELECT ON `hylson`.`hylson_livro` TO `mariasilva`@`%`;
GRANT SELECT, UPDATE ON `hylson`.`hylson_pessoa` TO `mariasilva`@`%`;

FLUSH PRIVILEGES;

show grants for 'mariasilva'@'%'

marcosoliveira / moliveira

CREATE USER 'marcosoliveira'@'%' IDENTIFIED BY 'moliveira';
GRANT SELECT ON `hylson`.`hylson_livros_emprestados` TO `moliveira`@`%`;

CREATE OR REPLACE
    DEFINER = 'root'@'%'
    SQL SECURITY DEFINER
VIEW hylson_livros_emprestados as
select nome, titulo from hylson_emprestimo he
join hylson_pessoa hp on he.pessoa = hp.id 
join hylson_livro hl on he.livro = hl.id
